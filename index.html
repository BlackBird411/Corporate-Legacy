// supabase/functions/player-profile-lookup/index.ts // // Reads a player's live ProfileStore data directly from Roblox's Open Cloud // DataStore API — works for ONLINE and OFFLINE players, since it reads the // saved DataStore entry rather than talking to a live game server. // // This powers the admin console's "Personnel Database" profile lookup: // credits, XP, inventory, join date (first-seen), all in one call, plus // this function returns the raw admin_actions history for the same player // so the frontend can render everything in a single request. // // DataStore name + key format MUST exactly match DataLoader.lua: // ProfileStore = ProfileService.New("PlayerData", dataTemplate) // key = "PlayerB_" .. player.UserId // // Secrets required (set once via `supabase secrets set`): // ROBLOX_API_KEY — Open Cloud key, needs datastore read scope // (universe-datastores.objects:read) for this universe // ROBLOX_UNIVERSE_ID — same universe ID used by trigger-admin-action // // SUPABASE_URL and SUPABASE_SERVICE_ROLE_KEY are injected automatically. // // IMPORTANT: "Enforce JWT Verification" must be OFF for this function, // same as trigger-admin-action and roblox-lookup — auth is handled // manually below via the admin_sessions token. // // Deploy with: supabase functions deploy player-profile-lookup import { createClient } from "https://esm.sh/@supabase/supabase-js@2"; const ROBLOX_API_KEY = Deno.env.get("ROBLOX_API_KEY")!; const ROBLOX_UNIVERSE_ID = Deno.env.get("ROBLOX_UNIVERSE_ID")!; const DATASTORE_NAME = "PlayerData"; const SUPABASE_URL = Deno.env.get("SUPABASE_URL")!; const SUPABASE_SERVICE_ROLE_KEY = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!; const corsHeaders = { "Access-Control-Allow-Origin": "*", // tighten to your domain once confirmed working "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type", "Access-Control-Allow-Methods": "POST, OPTIONS", }; Deno.serve(async (req: Request) => { if (req.method === "OPTIONS") { return new Response(null, { headers: corsHeaders }); } if (req.method !== "POST") { return json({ error: "Method not allowed" }, 405); } try { const adminClient = createClient(SUPABASE_URL, SUPABASE_SERVICE_ROLE_KEY); // --------------------------------------------------------- // 1. Authenticate — identical session check to every other // admin function, kept in sync intentionally. // --------------------------------------------------------- const authHeader = req.headers.get("Authorization") || ""; const token = authHeader.replace(/^Bearer\s+/i, "").trim(); if (!token) return json({ error: "Missing Authorization header" }, 401); const { data: session, error: sessionError } = await adminClient .from("admin_sessions") .select("admin_id, expires_at") .eq("token", token) .single(); if (sessionError || !session) return json({ error: "Invalid or expired session" }, 401); if (new Date(session.expires_at).getTime() < Date.now()) { return json({ error: "Session expired, please sign in again" }, 401); } // --------------------------------------------------------- // 2. Validate request body // --------------------------------------------------------- const body = await req.json(); const robloxUserId = Number(body?.roblox_user_id); if (!robloxUserId || !Number.isFinite(robloxUserId)) { return json({ error: "roblox_user_id is required" }, 400); } const entryKey = `PlayerB_${robloxUserId}`; // --------------------------------------------------------- // 3. Fetch the DataStore entry from Open Cloud. // GET /v1/universes/{universeId}/standard-datastores/datastore/entries/entry // --------------------------------------------------------- const dsUrl = `https://apis.roblox.com/datastores/v1/universes/${ROBLOX_UNIVERSE_ID}` + `/standard-datastores/datastore/entries/entry` + `?datastoreName=${encodeURIComponent(DATASTORE_NAME)}` + `&entryKey=${encodeURIComponent(entryKey)}`; const dsResponse = await fetch(dsUrl, { method: "GET", headers: { "x-api-key": ROBLOX_API_KEY }, }); let profileFound = true; let entry: any = null; if (dsResponse.status === 404) { // No DataStore entry yet — valid outcome, e.g. UserId typo'd, or a // real Roblox account that has genuinely never joined this game. profileFound = false; } else if (!dsResponse.ok) { const errText = await dsResponse.text(); console.error(`Roblox DataStore API ${dsResponse.status}: ${errText}`); return json({ error: "Roblox DataStore lookup failed" }, 502); } else { entry = await dsResponse.json(); } // entry.Data / entry.MetaData mirror the { Data, MetaData, GlobalUpdates } // wrapper ProfileStore itself writes — see ProfileStore.lua's // UpdateAsync/MissingProfileHandle for the exact shape. const profileData = entry?.Data ?? null; const metaData = entry?.MetaData ?? null; // --------------------------------------------------------- // 4. Pull this player's admin action history from our own // audit table, same data "Recent actions" already shows, // just filtered to one player so the profile view is complete // in a single request. // --------------------------------------------------------- const { data: actions, error: actionsError } = await adminClient .from("admin_actions") .select("*") .eq("roblox_user_id", robloxUserId) .order("created_at", { ascending: false }) .limit(50); if (actionsError) { console.error("Failed to load admin_actions:", actionsError); // Non-fatal — still return the profile data, just with empty history. } // --------------------------------------------------------- // 5. Shape the response for the frontend. // --------------------------------------------------------- return json({ roblox_user_id: robloxUserId, profile_found: profileFound, first_joined: metaData?.ProfileCreateTime ?? null, // Unix seconds, same value as profile.FirstSessionTime in-game credits: profileData?.Credits ?? null, xp: profileData?.XP ?? null, total_xp: profileData?.TotalXP ?? null, total_time_seconds: profileData?.TotalTime ?? null, skill_points: profileData?.SkillPoints ?? null, bank: profileData?.Bank ?? null, inventory: profileData?.Inventory ?? null, stock_portfolio: profileData?.StockPortfolio ?? null, admin_actions: actions ?? [], }); } catch (err) { console.error("Unhandled error:", err); return json({ error: "Internal server error" }, 500); } }); function json(body: unknown, status = 200): Response { return new Response(JSON.stringify(body), { status, headers: { ...corsHeaders, "Content-Type": "application/json" }, }); }
